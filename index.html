Conectar ao Supabase

Informe as credenciais do seu projeto Supabase. Os dados ficam salvos no navegador.

📋 SQL para criar as tabelas (clique para expandir) 
-- Execute no SQL Editor do seu projeto Supabase

create table employees (
  id uuid primary key default gen_random_uuid(),
  name text not null,
  pin varchar(6) not null,
  is_admin boolean default false,
  jornada_horas numeric default 8,
  created_at timestamptz default now()
);

create table time_records (
  id uuid primary key default gen_random_uuid(),
  employee_id uuid references employees(id) on delete cascade,
  date date not null,
  entrada timestamptz,
  inicio_intervalo timestamptz,
  fim_intervalo timestamptz,
  saida timestamptz,
  horas_trabalhadas numeric,
  hora_extra numeric default 0,
  observacao text,
  created_at timestamptz default now(),
  unique(employee_id, date)
);

create table absences (
  id uuid primary key default gen_random_uuid(),
  employee_id uuid references employees(id) on delete cascade,
  date date not null,
  justificativa text not null,
  tipo text default 'falta',
  created_at timestamptz default now(),
  unique(employee_id, date)
);

-- Desabilitar RLS (ferramenta interna)
alter table employees disable row level security;
alter table time_records disable row level security;
alter table absences disable row level security;
